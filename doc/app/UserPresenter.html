def profile_picture(type = "")
  if user.profile_picture.present?
    if type == :thumb
      image_tag user.profile_picture.thumb
    else
      image_tag user.profile_picture
    end
  else
    image_tag "default.png", :size => type == :thumb ? "50x50" : ""
  end
end
